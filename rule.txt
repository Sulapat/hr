3. 🏗️ backend/Dockerfile — Build แบบ Production
dockerfileFROM node:20-slim AS builder
RUN apt-get update -y && apt-get install -y openssl
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build

FROM node:20-slim
RUN apt-get update -y && apt-get install -y openssl
WORKDIR /app
COPY package*.json ./
RUN npm install --omit=dev
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/prisma ./prisma
EXPOSE 3001
CMD ["npm", "run", "start"]

4. 🌐 frontend/Dockerfile — Build Static แทน Dev Server
dockerfileFROM node:20-slim AS builder
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build

FROM nginx:alpine
COPY --from=builder /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
และสร้างไฟล์ frontend/nginx.conf:
nginxserver {
    listen 80;
    root /usr/share/nginx/html;
    index index.html;

    # รองรับ React Router
    location / {
        try_files $uri $uri/ /index.html;
    }

    # Proxy API ไป backend
    location /api/ {
        proxy_pass http://backend:3001;
    }

    location /uploads/ {
        proxy_pass http://backend:3001;
    }
}

5. 📦 วิธี Deploy บน Synology
bash# 1. copy โฟลเดอร์ไปไว้ใน Synology เช่น /volume1/docker/hr-system

# 2. SSH เข้า NAS แล้วรัน
cd /volume1/docker/hr-system
docker compose up -d --build

# 3. Migrate ฐานข้อมูล
docker compose exec backend npx prisma migrate deploy
docker compose exec backend npm run db:seed
